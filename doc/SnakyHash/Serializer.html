  # :nocov:

  unless VersionGem::Ruby.gte_minimum_version?("2.4")
    base.include(BackportedInstanceMethods)
  end
  # :nocov:
end

def load(raw_hash)

  hash = load_hash(hash)
  new(hash)
end
